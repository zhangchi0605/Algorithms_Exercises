## 216.组合总和III
```cpp
class Solution {
private:
    vector<vector<int>> result;
    vector<int> path;
    int         sum{0};
public:
    void backtracking(int k, int n, int start_index)
    {
        if (path.size() == k)
        {
            if (sum == n)
            {
                result.push_back(path);
            }
            return;
        }

        for (int i = start_index; i <= 9 - (k - path.size()) + 1; i++)
        {
            path.push_back(i);
            sum += i;
            backtracking(k, n, i+1);
            path.pop_back();
            sum -= i;
        }
    }

    vector<vector<int>> combinationSum3(int k, int n) {
        backtracking(k, n, 1);
        return result;
    }
};
```
